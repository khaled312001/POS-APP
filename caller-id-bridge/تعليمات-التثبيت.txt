================================================
  FRITZ!Card — ربط معرّف المتصل بنظام POS
  تعليمات التثبيت على كمبيوتر المطعم
================================================

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  ما تحتاجه:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ✓ جهاز FRITZ!Card USB v2.1 متصل بالكمبيوتر
  ✓ كابل ISDN متوصل بخط التليفون
  ✓ Node.js مثبت على الكمبيوتر (nodejs.org)
  ✓ الأنترنت شغّال على الكمبيوتر

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  الخطوة 1 — تثبيت Node.js (مرة واحدة فقط)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. روح موقع: https://nodejs.org
  2. حمّل النسخة LTS (الخضراء)
  3. ثبّته بكل الإعدادات الافتراضية
  4. أعد تشغيل الكمبيوتر

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  الخطوة 2 — تثبيت الدرايفرات (مرة واحدة فقط)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. ابحث Google عن: AVM FRITZ!Card USB v2.1 driver
  2. حمّل الدرايفر من موقع AVM: https://avm.de
  3. ثبّته كـ Administrator وأعد التشغيل
  4. تأكد إن الجهاز ظاهر في Device Manager بدون علامة تعجب

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  الخطوة 3 — وضع المجلد على الكمبيوتر
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  انقل المجلد الكامل "caller-id-bridge" إلى مكان ثابت، مثل:

      C:\POS\caller-id-bridge\

  لا تحرّك المجلد بعد التثبيت.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  الخطوة 4 — التشغيل التلقائي مع Windows
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. كليك يمين على ملف "install-autostart.bat"
  2. اختر "Run as administrator" (تشغيل كمسؤول)
  3. لو سأل عن UAC → اضغط YES
  4. لما يسأل "هل تريد تشغيل البريدج الآن؟" → اكتب Y واضغط Enter

  ✓ خلاص! البريدج هيشتغل تلقائياً مع كل تشغيل Windows
  ✓ يعيد التشغيل لوحده لو وقع (auto-restart)
  ✓ يشتغل في الخلفية بدون نافذة سوداء

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  الخطوة 5 — اختبار الربط
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  افتح التطبيق على المتصفح:
      https://pos.barmagly.tech/app/

  ادخل على Settings (الإعدادات) → Caller ID
  اضغط "Simulate Incoming Call"

  يجب أن يظهر popup في صفحة POS فوراً بـ:
      ✓ تمت الإضافة تلقائياً
      + رقم الهاتف
      + اسم العميل (لو موجود في النظام)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  كيف يعمل النظام؟
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  تليفون بيرن
       ↓
  FRITZ!Card بتحس بالمكالمة وتمسك رقم المتصل
       ↓
  البريدج بيبعت الرقم للسيرفر (pos.barmagly.tech)
       ↓
  السيرفر بيبحث عن العميل بالرقم في قاعدة البيانات
       ↓
  صفحة POS في المتصفح بتضيف العميل في الفاتورة تلقائياً
  (في أقل من ثانية)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  مشاكل شائعة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  المشكلة: "Cannot load capi2032.dll"
  الحل:    الدرايفرات مش مثبتة → راجع الخطوة 2

  المشكلة: "FRITZ!Card not connected"
  الحل:    تأكد إن الجهاز متوصل بـ USB وكابل ISDN متوصل

  المشكلة: مش بيظهر popup في المتصفح
  الحل:    تأكد إن المتصفح مفتوح على pos.barmagly.tech/app/
            وإنك مسجّل دخول

  المشكلة: "Unauthorized"
  الحل:    راجع ملف config.json إن الـ secret صح

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  الملفات الموجودة في المجلد
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  config.json           ← إعدادات الربط (السيرفر، التيننت، السر)
  capi-bridge.js        ← برنامج الربط الرئيسي (لا تعدّله)
  start.bat             ← تشغيل يدوي مع auto-restart
  install-autostart.bat ← تثبيت التشغيل التلقائي مع Windows
  uninstall-autostart.bat ← إلغاء التشغيل التلقائي
  _launcher.vbs         ← مشغّل صامت في الخلفية (لا تحذفه)
